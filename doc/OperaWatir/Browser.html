def self.settings=(settings={})
  @opera_driver_settings = nil # Bust cache
  @settings = settings.merge! :launcher => ENV['OPERA_LAUNCHER'] || '',
                              :path     => ENV['OPERA_PATH'] || '',
                              :args     => ENV['OPERA_ARGS'] || ''
end
