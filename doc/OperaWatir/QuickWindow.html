def window_info_string
  "    Name: " + name +
  "\n   Title: " + title +
  "\n      ID: " + id.to_s +
  "\n    Type: " + type.to_s +
  "\nOnScreen: " + on_screen?.to_s +
  "\n     Pos: x=" + element.getRect().x.to_s + ", y=" + element.getRect().y.to_s +
  "\n    Size: width=" + element.getRect().width.to_s + ", height=" + element.getRect().height.to_s + "\n"
end
