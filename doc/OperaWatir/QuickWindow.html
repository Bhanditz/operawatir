def print_window_info
  puts "    Name: " + name
  puts "   Title: " + title
  puts "      ID: " + id.to_s
  puts "    Type: " + type.to_s
  puts "OnScreen: " + on_screen?.to_s
  puts "     Pos: x=" + element.getRect().x.to_s + ", y=" + element.getRect().y.to_s
  puts "    Size: width=" + element.getRect().width.to_s + ", height=" + element.getRect().height.to_s
  puts ""
end
